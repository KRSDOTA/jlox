expression      -> block;
block		    -> equality ("," equality)* ;
equality        -> comparison ( term ( "==" | "!=" ) comparison )* ;
comparison      -> term ( ( ">" | ">=" | "<" | "<=" ) term )* ;
term            -> factor (factor ("-" | "+" ) factor)* ;
factor          -> unary ( ( "/" | "*" ) unary )* ;
unary           -> ( "!" | "-" ) unary
                -> | primary ;
primary         -> NUMBER | STRING | "true" | "false" | "nil"
               | "(" expression ")" ;
